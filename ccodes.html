        *Write A C Program To swap 2 values*
#include<stdio.h>
main()
{
    int a,b,temp;
    printf("Enter The Number");
    scanf("%d%d",&a,&b);
    temp=a;
    a=b;
    b=temp;
    printf("Swapped Values are:");
    printf("a=%d,b=%d",a,b);
}
    
        *Write a C Program To find area and circumference of a circle*
#include<stdio.h>
main()
{
    float radius,area,perimeter;
    printf("Enter The Radius of the Circle:");
    scanf("%f",&radius);
    area=3.14*radius*radius;
    perimeter=2*3.14*radius;
    printf("\n\nThe area of the circle with radius %f is %f",radius,area);
    printf("\n\nThe Circumference of the circle is %f",perimeter);
    getch();
}
    
        *Write A C Program to Find Whether The Given Number Is Odd Or Even*
#include<stdio.h>
int main()
{
    int number;
    printf("Enter the number: ");
    scanf("%d",&number);
    if(number%2==0)
        printf("Number %d is even",number);
    else
        printf("Number %d is odd",number);
    return 0;
}
